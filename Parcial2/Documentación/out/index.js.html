/**
 * Funcion para obtener el area de un circulo
 * @param {number} radio radio de la circunferencia
 * @returns {number} area del circulo
 */
function calcularAreaCirculo(radio) {
    const pi = Math.PI;
    const area = pi * radio ** 2;
    return area;
}
